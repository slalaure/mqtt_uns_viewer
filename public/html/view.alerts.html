// Topic: spBv1.0/stark_industries/DDATA/robot_arm_01 // Find specific metric in array if (!msg.payload.metrics) return false; const metric = msg.payload.metrics.find(m => m.name === "Motor/Temp"); // Trigger if temp > 80.0 return metric && metric.value > 80.0;
3. Anomaly Detection (Compare vs History)
// Topic: stark_industries/malibu_facility/bms/main_power_feed // Compare current value vs Average of last 10 values const currentVal = msg.payload.value; // SQL to get avg from history (DuckDB syntax) const sql = ` SELECT AVG(CAST(payload->>'value' AS DOUBLE)) as avg_val FROM korelate_events WHERE topic = '${msg.topic}' ORDER BY timestamp DESC LIMIT 10 `; // Async database call const result = await db.get(sql); // Trigger if current value is 20% higher than average if (result && result.avg_val) { return currentVal > (result.avg_val * 1.20); } return false;
